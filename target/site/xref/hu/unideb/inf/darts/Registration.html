6   package hu.unideb.inf.darts;
7   
8   import java.io.File;
9   import java.io.IOException;
10  import java.util.ArrayList;
11  import javax.xml.bind.JAXBContext;
12  import javax.xml.bind.JAXBException;
13  import javax.xml.bind.Marshaller;
14  import org.slf4j.LoggerFactory;
15  import org.xml.sax.SAXException;
16  
17  /**
18   * A regisztrációt végző osztály.
19   * @author Ak
20   */
21  public class Registration {
22      private static final org.slf4j.Logger logger = LoggerFactory.getLogger(Registration.class);
23      private Users users = new Users();
24      private boolean ch1 = false;
25      private String invalidName;
26      private CreateFolder cf = new CreateFolder();
27      private Reader rd = new Reader();
28  
29      /**
30       * Visszaadja nem érvényes név esetén adott üzenetet.
31       * @return invalidName {@code String}
32       */
33      public String getInvalidName() {
34          return invalidName;
35      }
36  
37      /**
38       * A regisztációt végző metódus.
39       * @param name Egy {@code User} neve.
40       * @throws JAXBException Írás és olvsás miatt
41       * @throws SAXException Írás és olvsás miatt
42       * @throws IOException Írás és olvsás miatt
43       */
44      public void registration(String name) throws JAXBException, SAXException, IOException {
45  
46          if (!name.isEmpty()) {
47              cf.create();
48              rd.read();
49              users.setUsers(new ArrayList<>());
50              users.setUsers(Reader.users.getUsers());
51              for (User user : users.getUsers()) {
52                  if (user.getName().equals(name)) {
53                      ch1 = true;
54                      invalidName = "Existing name";
55                      logger.warn("Létező név");
56                      break;
57                  }else{
58                      ch1 = false;
59                  }
60              }
61              if (!ch1) {
62                  users.getUsers().add(new User(name));
63                      invalidName = "Registered";
64                      write();
65                      logger.warn("Regisztáció sikeres");
66              }
67  
68          } else {
69              invalidName = "Invalid Name";
70              logger.warn("Szabálytalan név");
71          }
72      }
73  
74      private void write() throws JAXBException {
75          JAXBContext jaxbContext = JAXBContext.newInstance(Users.class);
76          Marshaller jaxbMarshaller = jaxbContext.createMarshaller();
77          jaxbMarshaller.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, true);
78          jaxbMarshaller.marshal(users, System.out);
79          jaxbMarshaller.marshal(users, new File(cf.getPath() + "darts.xml"));
80      }
81  
82  }
